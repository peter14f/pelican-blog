698-partition_to_k_equal_sum_subsets

Mar 19 2018 Tags DFS / DFS with backtracking / Recursion

LeetCode Problem Link

For most DFS with backtracking problems, you maintain one path. This problem requires maintaining k paths.

public boolean canPartitionKSubsets(int[] nums, int k) {
    if (nums.length < k) {
        return false;
    }

    int total = 0;
    for (int num : nums) {
        total += num;
    }

    if (total % k != 0) {
        return false;
    }

    Arrays.sort(nums);
    int target = total/k;

    return dfsCanPartitionKSubsets(nums, nums.length - 1, new int[k], target);
}

private boolean dfsCanPartitionKSubsets(int[] nums, int index, int[] sums, int target) {
    if (index < 0) {
        for (int i=0; i<sums.length; i++) {
            if (sums[i] != target) {
                return false;
            }
        }
        return true;
    }

    // having a single element larger than target means k nonempty equal subsets is impossible
    if (nums[index] > target) {
        return false;
    }

    for (int i=0; i<sums.length; i++) {
        if (sums[i] + nums[index] <= target) {
            sums[i] += nums[index];
            if (dfsCanPartitionKSubsets(nums, index-1, sums, target)) {
                return true;
            }
            sums[i] -= nums[index]; // including nums[index] in set i
                                    // does not result in k equal partitions
        }
    }
    return false;
}
