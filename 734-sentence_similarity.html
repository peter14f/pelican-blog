734-sentence_similarity

Mar 20 2018

LeetCode Problem Link

Use a Map<String, Set<String>> similar to look up similar words.

public boolean areSentencesSimilar(String[] words1, String[] words2, String[][] pairs) {
    if (words1.length != words2.length) {
        return false;
    }

    Map<String, Set<String>> similar = new HashMap<>();
    if (pairs != null) {
        for (String[] pair : pairs) {
            String w1 = pair[0];
            String w2 = pair[1];

            if (!similar.containsKey(w1)) {
                similar.put(w1, new HashSet<>());
            }
            similar.get(w1).add(w2);

            if (!similar.containsKey(w2)) {
                similar.put(w2, new HashSet<>());
            }
            similar.get(w2).add(w1);
        }
    }

    for (int i=0; i<words1.length; i++) {
        String w1 = words1[i];
        String w2 = words2[i];

        if (w1.equals(w2)) {
            continue;
        }
        if (!similar.containsKey(w1)) {
            return false;
        }
        if (!similar.get(w1).contains(w2)) {
            return false;
        }
    }
    return true;
}
