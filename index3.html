739-daily_temperatures

Jul 10 2018 Tags Stacks

LeetCode Problem Link

The trivial solution would take O(n2) time. This problem reminds of 84-largest_rectangle.

public int[] dailyTemperatures2(int[] temperatures) {
    Stack<Integer> stk = new Stack<Integer>(); // stores the index
    int[] ans = new int[temperatures.length];
    int i=0;
    while (!stk.isEmpty() || i<temperatures.length) {
        if (stk.isEmpty()) {
            stk.push(i);
            i++;
            continue;
        }

        if (i >= temperatures.length || temperatures[i] > temperatures[stk.peek()]) {
            int index = stk.pop();
            if (i >= temperatures.length) {
                ans[index] = 0;
            } else {
                ans[index] = i-index;
            }
        } else {
            stk.push(i);
            i++;
        }
    }
    return ans;
}
