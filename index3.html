461-hamming_distance

Mar 17 2018 Tags Couting Bits / XOR / Bit Manipulation

public int hammingDistance(int x, int y) {
    int z = x ^ y;
    int hammingDistance = 0;
    for (int i=0; i<32; i++) {
        if ((z & 1) == 1) {
            hammingDistance++;
        }
        z = z >>> 1;
    }
    return hammingDistance;
}
