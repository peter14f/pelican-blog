744-minimize_max_distance_to_gas_station

Mar 20 2018 Tags Heap / Binary Search

LeetCode Problem Link

Greedy will not work. Take this input [1, 5, 10] When k=1, ans=4 (We put the one station right at 7.5 When k=2, ans=2.5 (One station at 7.5 and another at 3) When k=3, ans=2 (One station at 3, one at 6.67, and last one at 5 + 5/3 + 5/3)

Using a heap will do it. Note that when we construct a new Interval, k is initialized to 1 and that means there is not stations between left and right. This solution is too slow for OJ.

public double minmaxGasDist(int[] stations, int K) {
    Arrays.sort(stations);
    int n = stations.length;

    PriorityQueue<Interval> pq = new PriorityQueue<>(Collections.reverseOrder());

    for (int i=1; i<n; i++) {
        Interval interval = new Interval(stations[i-1], stations[i]);
        //System.out.println(interval);
        pq.offer(interval);
    }

    for (int i=0; i<K; i++) {
        Interval interval = pq.poll();
        interval.k++;
        pq.offer(interval);
    }

    Interval interval = pq.peek();
    return interval.getDist();
}

static class Interval implements Comparable<Interval> {
    double left;
    double right;
    int k;

    public Interval(double l, double r) {
        this.left = l;
        this.right = r;
        this.k = 1;
    }

    public double getDist() {
        return (this.right - this.left)/this.k;
    }

    @Override
    public int compareTo(Interval o) {
        double myD = this.getDist();
        double otherD = o.getDist();

        if (myD > otherD) {
            return 1;
        } else if (myD < otherD) {
            return -1;
        } else {
            return 0;
        }
    }

    @Override
    public String toString() {
        return this.left + " - " + this.right;
    }
}

Another way is to use binary search to guestimate what the distance would be. The hint says that answer within the range 10^6 will be accepted...

public double minmaxGasDist(int[] stations, int K) {
    Arrays.sort(stations);
    int maxDistance = 0;
    int n = stations.length;
    for (int i=1; i<n; i++) {
        maxDistance = Math.max(maxDistance, stations[i] - stations[i-1]);
    }

    double min = 0;
    double max = maxDistance;

    while (max > min + 0.000001) {
        double middle = min + (max - min) / 2;

        int numStations = getNumStations(stations, middle);
        if (numStations > K) {
            // we need to increase d to decrease the number of stations
            min = middle;
        } else {
            // we need to decrease d to increase the number of stations
            max = middle;
        }
    }

    return min + (max - min) / 2;
}

private int getNumStations(int[] stations, double d) {
    int sum = 0;
    for (int i=1; i<stations.length; i++) {
        double distance = stations[i] - stations[i-1];

        // ex: dist = 4
        //     d = 2
        //     ceil(4/2) - 1 = 1
        //
        // ex: dist = 5
        //     d = 2
        //     ceil(5/2) - 1 = 2
        int num = (int) (Math.ceil(distance / d) - 1);
        sum += num;
    }
    return sum;
}
