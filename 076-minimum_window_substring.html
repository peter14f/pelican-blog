076-minimum_window_substring

Nov 07 2015 Tags Two Pointers

LeetCode Problem Link

Note that there is a simplier version of the problem which t only contains distinct characters. We do use two pointers, but they don't start at two different ends. Instead the start and end pointers both start at index 0. We advance end all the way until a solution is found and then attempt to advance start to find a shorter/better solution. Let me first post the solution to the simplier version of the problem. Note that once a solution is found, we would advance start and decrement numChars so that end should be further advanced to look for the one missing character to find the next solution.

public String shortestSubstringSimple(String s, String t) {
    // set of characters in t - assume each character shows up at most once in t
    Set<Character> set = new HashSet<>();
    for (int i=0; i<t.length(); i++) {
        set.add(t.charAt(i));
    }

    // both start and end only advance forward
    // the current window is [start, end]
    int end=0;
    int start=0;
    Map<Character, Integer> cnt = new HashMap<>();
    // number of characters in t is currently in the current window
    int numChars = 0;
    int min = s.length() + 1; // even if the entire string is the answer, that will be shorter than min's initail value
    String shortest = "";

    while (end < s.length()) {
        char c = s.charAt(end);

        if (set.contains(c)) {
            if (cnt.containsKey(c)) {
                cnt.put(c, 1 + cnt.get(c));
            } else {
                cnt.put(c, 1);
                numChars++;
            }
        }

        if (numChars == t.length()) {
            // an answer is found s.substring(start, end+1)
            // but we could potentially do better by advancing start

            // we advance start if it's an irrelevant character or if
            // the current substring has more than one occurrence
            char startC = s.charAt(start);
            while (!set.contains(startC) || cnt.get(startC) > 1) {
                if (set.contains(startC)) {
                    cnt.put(startC, cnt.get(startC) - 1);
                }
                start++;
                startC = s.charAt(start);
            }

            int len = end - start + 1;
            if (len < min) {
                shortest = s.substring(start, end+1);
                min = len;
            }

            // advance start once more to make the window miss one
            // char from 's', which will be the next char 'end' should be
            // looking for
            start++;
            numChars--;
            cnt.remove(startC);
        }

        end++;
    }

    return shortest;
}

Now let's go back to the original problem. Now a character may show up more than once in t. We would solve it pretty much in the same way, but using a Set<Character> is no longer enough. We will instead build a Map<Character, Integer> and call it tMap.

Let the length of s be n. In the worst case, t only contains 1 character and it appears only once in s and is the last character of s. end goes from 0 to n-1, then start goes from 0 to n-1 as well. Therefore this takes O(n) time.

public String minWindow(String s, String t) {

    char[] arrT = t.toCharArray();
    char[] arrS = s.toCharArray();

    HashMap<Character, Integer> cnt = new HashMap<Character, Integer>();

    // how many times each character in t appears in t
    for (int i=0; i<arrT.length; i++) {
        if (cnt.containsKey(arrT[i])) {
            cnt.put(arrT[i], cnt.get(arrT[i])+1);
        }
        else {
            cnt.put(arrT[i], 1);
        }
    }

    // check if we've seen all characters in t
    HashMap<Character, Integer> curCnt = new HashMap<Character, Integer>();

    int end = 0;
    int numCharSeen = 0;
    int start = 0;
    int minLength = Integer.MAX_VALUE;
    int minStart = -1;

    while (end < arrS.length) {

        char c = arrS[end];

        // c is in 't'
        if (cnt.containsKey(c)) {

            if (curCnt.containsKey(c)) {
                if (curCnt.get(c) < cnt.get(c))
                    numCharSeen++;

                curCnt.put(c, curCnt.get(c) + 1);
            }
            else {
                curCnt.put(c, 1);
                numCharSeen++;
            }
        }

        if (numCharSeen == arrT.length) {
            char startC = arrS[start];

            // this while loop moves 'start' forward
            while (!cnt.containsKey(startC) || curCnt.get(startC) > cnt.get(startC)) {

                if (cnt.containsKey(startC)) {
                    // moving start forward means the window has one fewer character
                    curCnt.put(startC, curCnt.get(startC) - 1);
                }

                start++;
                startC = arrS[start];
            }

            int length = end-start+1;
            if (length < minLength) {
                minLength = length;
                minStart = start;
            }
        }

        end++;
    }

    if (minStart == -1)
        return "";
    else
        return new String(arrS, minStart, minLength);
}
