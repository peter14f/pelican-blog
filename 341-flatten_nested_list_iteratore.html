341-flatten_nested_list_iterator

Mar 17 2018 Tags Couting Bits / Bit Manipulation

LeetCode Problem Link

The easiest way is recursively flatten the nested list at constructor time.

public class NestedIterator implements Iterator<Integer> {
    List<Integer> l;
    Iterator<Integer> itr;
    public NestedIterator(List<NestedInteger> nestedList) {
        this.l = new LinkedList<>();
        recursivelyFlatten(nestedList, this.l);
        this.itr = this.l.iterator();
    }

    private void recursivelyFlatten(List<NestedInteger> nestedList, List<Integer> l) {
        for (int i=0; i<nestedList.size(); i++) {
            if (nestedList.get(i).isInteger()) {
                l.add(nestedList.get(i).getInteger());
            } else {
                recursivelyFlatten(nestedList.get(i).getList(), l);
            }
        }
    }

    @Override
    public Integer next() {
        return this.itr.next();
    }

    @Override
    public boolean hasNext() {
        return this.itr.hasNext();
    }

    interface NestedInteger {
        // @return true if this NestedInteger holds a single integer, rather than a nested list.
        public boolean isInteger();
        // @return the single integer that this NestedInteger holds, if it holds a single integer
        // Return null if this NestedInteger holds a nested list
        public Integer getInteger();
        // @return the nested list that this NestedInteger holds, if it holds a nested list
        // Return null if this NestedInteger holds a single integer
        public List<NestedInteger> getList();
    }
}

Another solution is to use a Stack and always maintain that the top of the stack is an int. Sure it delays the flatten operation but it doesn't actually use less space...

public class NestedIterator implements Iterator<Integer> {
    Stack<NestedInteger> stk;
    public NestedIterator(List<NestedInteger> nestedList) {
        this.stk = new Stack<>();
        for (int i=nestedList.size()-1; i>=0; i--) {
            this.stk.push(nestedList.get(i));
        }
    }

    @Override
    public Integer next() {
        hasNext();
        return this.stk.pop().getInteger();
    }

    @Override
    public boolean hasNext() {
        if (this.stk.isEmpty()) {
            return false;
        }

        while (!stk.peek().isInteger()) {
            NestedInteger item = stk.pop();
            List<NestedInteger> l = item.getList();
            for (int i=l.size() - 1; i>=0; i--) {
                this.stk.push(l.get(i));
            }
            if (this.stk.isEmpty()) {
                break;
            }
        }

        return !this.stk.isEmpty();
    }

    interface NestedInteger {
        // @return true if this NestedInteger holds a single integer, rather than a nested list.
        public boolean isInteger();
        // @return the single integer that this NestedInteger holds, if it holds a single integer
        // Return null if this NestedInteger holds a nested list
        public Integer getInteger();
        // @return the nested list that this NestedInteger holds, if it holds a nested list
        // Return null if this NestedInteger holds a single integer
        public List<NestedInteger> getList();
    }

    }
}
