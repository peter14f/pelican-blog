536-construct_binary_tree_from_string

Mar 18 2018 Tags Binary Trees / Deserialization

LeetCode Problem Link

Use a Stack<TreeNode> to store parent nodes, when we encounter a ')', then we pop() from the stack. Encountering a '(' means we need to push a new node onto the stack.

/**
 * There will only be '(', ')', '-' and '0' ~ '9' in the input string.
 *
 * An empty tree is represented by "" instead of "()".
 */
public TreeNode str2Tree(String s) {
    if (s.length() == 0) {
        return null;
    }

    TreeNode root = null;
    Stack<TreeNode> stk = new Stack<>();

    s = '(' + s + ')';
    int n = s.length();

    for (int i=0; i<n; i++) {
        if (s.charAt(i) == ')') {
            stk.pop();
        } else if (s.charAt(i) == '(') {
            int j = i+1;
            int sum = 0;
            boolean isNegative = false;
            if (s.charAt(j) == '-') {
                isNegative = true;
                j++;
            }
            while (s.charAt(j) >= '0' && s.charAt(j) <= '9') {
                sum = sum * 10 + (s.charAt(j) - '0');
                j++;
            }

            TreeNode newNode = new TreeNode((isNegative)?-sum:sum);
            if (root == null) {
                root = newNode;
            }
            if (!stk.isEmpty()) {
                TreeNode parent = stk.peek();
                if (parent.left == null) {
                    parent.left = newNode;
                } else {
                    parent.right = newNode;
                }
            }

            stk.push(newNode);

            i = j - 1; // for loop will add one to i at the end of this iteration
                       // next char expected is '(' or ')'
        }
    }

    return root;
}
