737-sentence_similarity_ii

Mar 20 2018 Tags Union Find / Graph / BFS / DFS

LeetCode Problem Link

Transitivity makes this into a graph problem. Can solve this using either DFS or BFS.

Also can be solved using Union Find.

    // values are the neighbors
    Map<String, Set<String>> similar = new HashMap<>();

    for (String[] pair : pairs) {
        String w1 = pair[0];
        String w2 = pair[1];

        if (!similar.containsKey(w1)) {
            similar.put(w1, new HashSet<>());
        }
        similar.get(w1).add(w2);

        if (!similar.containsKey(w2)) {
            similar.put(w2, new HashSet<>());
        }
        similar.get(w2).add(w1);
    }

    for (int i=0; i<words1.length; i++) {
        String w1 = words1[i];
        String w2 = words2[i];

        if (w1.equals(w2)) {
            continue;
        }

        if (!similar.containsKey(w1)) {
            // w1 has no neighbors...
            return false;
        }

        // now it's time to do bfs
        Set<String> visited = new HashSet<String>();

        if (!foundViaBFS(similar, w1, visited, w2)) {
            return false;
        }
    }

    return true;
}

private boolean foundViaBFS(Map<String, Set<String>> graph, String start,
        Set<String> visited, String end) {

    Queue<String> q = new LinkedList<String>();
    q.offer(start);
    visited.add(start);

    while (!q.isEmpty()) {
        String s = q.poll();
        if (s.equals(end)) {
            return true;
        }
        Set<String> neighbors = graph.get(s);
        for (String n : neighbors) {
            if (visited.contains(n)) {
                continue;
            }
            q.offer(n);
            visited.add(n);
        }
    }
    return false;
}
