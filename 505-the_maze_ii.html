Mar 17 2018 Tags Dijkstra / Graph / Graph Problem With A Spin

LeetCode Problem Link

A good practice for implementing Dijkstra's algorithm.

// in this problem, we are not trying to figure out the path
// we are only interested in knowing the shortest distance from
// start to destination... This should yell Dijkstra for you.
public int shortestDistance(int[][] maze, int[] start, int[] destination) {
    int m = maze.length;
    if (m < 1) {
        return -1;
    }
    int n = maze[0].length;

    // distant from start to all cells
    // not found in map -> distance is infinity
    Map<Cell, Integer> dist = new HashMap<>();

    // shortest distance from start has been figured out
    Set<Cell> done = new HashSet<>();

    PriorityQueue<CellAndDist> pq = new PriorityQueue<>();

    Cell startCell = new Cell(start[0], start[1]);
    pq.offer(new CellAndDist(startCell, 0));
    dist.put(startCell, 0);
    Cell destCell = new Cell(destination[0], destination[1]);

    while (!pq.isEmpty()) {
        CellAndDist curWd = pq.poll();

        // in our case, the graph may be partitioned
        // polling a infinity out means we're done
        if (curWd.d == Integer.MAX_VALUE) {
            break;
        }

        Cell cur = curWd.c;
        done.add(cur); // figured out the shortest distance from src to cur

        if (cur.equals(destCell)) {
            break;
        }

        // UP
        Cell up = getUpNeighbor(maze, cur.x, cur.y);
        if (up != null) {
            int distCurToUp = cur.x - up.x;
            int distStartToUp = (!dist.containsKey(up)) ? Integer.MAX_VALUE : dist.get(up);
            if (dist.get(cur) + distCurToUp < distStartToUp) {
                dist.put(up, dist.get(cur) + distCurToUp);
                CellAndDist upAndD = new CellAndDist(up, dist.get(cur) + distCurToUp);
                pq.remove(upAndD);
                pq.offer(upAndD);
            }
        }
        // RIGHT
        Cell right = getRightNeighbor(maze, cur.x, cur.y, n);
        if (right != null) {
            int distCurToRight = right.y - cur.y;
            int distStartToRight = (!dist.containsKey(right))? Integer.MAX_VALUE :dist.get(right);
            if (dist.get(cur) + distCurToRight < distStartToRight) {
                dist.put(right, dist.get(cur) + distCurToRight);
                CellAndDist rightAndD = new CellAndDist(right, dist.get(cur) + distCurToRight);
                pq.remove(rightAndD);
                pq.offer(rightAndD);
            }
        }
        // DOWN
        Cell down = getDownNeighbor(maze, cur.x, cur.y, m);
        if (down != null) {
            int distCurToDown = down.x - cur.x;
            int distStartToDown = (!dist.containsKey(down)) ? Integer.MAX_VALUE : dist.get(down);
            if (dist.get(cur) + distCurToDown < distStartToDown) {
                dist.put(down, dist.get(cur) + distCurToDown);
                CellAndDist downAndD = new CellAndDist(down, dist.get(cur) + distCurToDown);
                pq.remove(downAndD);
                pq.offer(downAndD);
            }
        }
        // LEFT
        Cell left = getLeftNeighbor(maze, cur.x, cur.y);
        if (left != null) {
            int distCurToLeft = cur.y - left.y;
            int distStartToLeft = (!dist.containsKey(left)) ? Integer.MAX_VALUE : dist.get(left);
            if (dist.get(cur) + distCurToLeft < distStartToLeft) {
                dist.put(left, dist.get(cur) + distCurToLeft);
                CellAndDist leftAndD = new CellAndDist(left, dist.get(cur) + distCurToLeft);
                pq.remove(leftAndD);
                pq.offer(leftAndD);
            }
        }
    }

    if (!dist.containsKey(destCell)) {
        return -1;
    } else {
        return dist.get(destCell);
    }
}

private Cell getLeftNeighbor(int[][] maze, int x, int y) {
    int newY = y;
    while (newY-1 >= 0 && maze[x][newY-1] == 0) {
        newY--;
    }
    if (newY == y) {
        return null;
    } else {
        return new Cell(x, newY);
    }
}

private Cell getDownNeighbor(int[][] maze, int x, int y, int m) {
    int newX = x;
    while (newX + 1 < m && maze[newX+1][y] == 0) {
        newX++;
    }
    if (newX == x) {
        return null;
    } else {
        return new Cell(newX, y);
    }
}

private Cell getRightNeighbor(int[][] maze, int x, int y, int n) {
    int newY = y;
    while (newY+1 < n && maze[x][newY+1] == 0) {
        newY++;
    }
    if (newY == y) {
        return null;
    } else {
        return new Cell(x, newY);
    }
}

private Cell getUpNeighbor(int[][] maze, int x, int y) {
    int newX = x;
    while (newX-1 >=0 && maze[newX-1][y] == 0) {
        newX--;
    }
    if (newX == x) {
        return null;
    } else {
        return new Cell(newX, y);
    }
}

static class CellAndDist implements Comparable<CellAndDist> {
    Cell c;
    int d;
    public CellAndDist(Cell c, int d) {
        this.c = c;
        this.d = d;
    }
    @Override
    public int compareTo(CellAndDist o) {
        return this.d - o.d;
    }
    @Override
    public int hashCode() {
        final int prime = 31;
        int result = 1;
        result = prime * result + ((c == null) ? 0 : c.hashCode());
        return result;
    }
    @Override
    public boolean equals(Object obj) {
        if (this == obj)
            return true;
        if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        CellAndDist other = (CellAndDist) obj;
        if (c == null) {
            if (other.c != null)
                return false;
        } else if (!c.equals(other.c))
            return false;
        return true;
    }
}

static class Cell {
    int x;
    int y;
    public Cell(int x, int y) {
        this.x = x;
        this.y = y;
    }
    @Override
    public int hashCode() {
        final int prime = 31;
        int result = 1;
        result = prime * result + x;
        result = prime * result + y;
        return result;
    }
    @Override
    public boolean equals(Object obj) {
        if (this == obj)
            return true;
        if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        Cell other = (Cell) obj;
        if (x != other.x)
            return false;
        if (y != other.y)
            return false;
        return true;
    }
    @Override
    public String toString() {
        return this.x + "," + this.y;
    }
}
