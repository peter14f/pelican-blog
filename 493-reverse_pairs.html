493-reverse_pairs

Mar 17 2018 Tags Merge Sort / Divide And Conquer / BIT / Self-Balancing BST

LeetCode Problem Link

This problem asks us to find the number of distinct (i, j) pairs such that - i < j - nums[i] > 2*nums[j]

The straightforward solution takes O(n^2) time.

public int reversePairs(int[] nums) {
    int n = nums.length;
    int ans = 0;
    for (int i=0; i<n; i++) {
        for (int j=i+1; j<n; j++) {
            long numI = nums[i];
            long numJ = nums[j];
            if (numI > 2*numJ) {
                ans++;
            }
        }
    }
    return ans;
}

When you do merge sort, the left array elements always come before the right array elements (in the original array). So if left[i] > 2*right[j], we know that number after i in the left array also satisfies the same condition if the left array is sorted.

public int reversePairs(int[] nums) {
    int[] tmp = new int[nums.length];
    int ans = mergeSort(nums, 0, nums.length - 1, tmp);
    return ans;
}

private int mergeSort(int[] nums, int start, int end, int[] tmp) {
    if (end <= start) {
        return 0;
    }

    int m = start + (end - start) / 2;
    int cntLeft = mergeSort(nums, start, m, tmp);
    int cntRight = mergeSort(nums, m+1, end, tmp);

    int i=start;
    int j=m+1;

    int cnt = cntLeft + cntRight;

    // count
    while (i<=m && j<=end) {
        long numI = nums[i];
        long numJ = nums[j];
        if (numI > 2*numJ) {
            // if numI satisfies the condition, then
            // all numbers after numI in the left array also satisfies
            // the condition
            cnt += (m-i+1);
            j++;
        } else {
            i++;
        }
    }

    // merge
    int k=start;
    i=start;
    j=m+1;
    while (i<=m && j<=end) {
        if (nums[i] <= nums[j]) {
            tmp[k] = nums[i];
            i++;
            k++;
        } else {
            tmp[k] = nums[j];
            j++;
            k++;
        }
    }

    while (i<=m) {
        tmp[k] = nums[i];
        i++;
        k++;
    }

    while (j<=end) {
        tmp[k] = nums[j];
        j++;
        k++;
    }

    for (int z=start; z<=end; z++) {
        nums[z] = tmp[z];
    }
    // nums[start...end] is now sorted
    return cnt;
}
