    List<List<Integer>> l;
    int row, col, m;

    public Vector2DB(List<List<Integer>> vec2d) {
        l = vec2d;
        m = l.size();
        row = 0;
        col = 0;
    }

    public int next() {
        int toReturn = l.get(row).get(col);
        col++;
        return toReturn;
    }

                return true;

    int bN = b.length();

    if (aN < bN)
        return true;

            break;
        }
