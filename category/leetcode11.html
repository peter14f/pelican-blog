263-ugly_number

Feb 19 2016 Tags Prime Factorization / Ugly Number

LeetCode Problem Link

We want to know if prime factors other than 2, 3, and 5 exist for num.

We can use while loops to keep dividing num by 2, 3, and 5 if it's divisible.

public boolean isUgly(int num) {
    if (num < 1)
        return false;

    while ((num % 2) == 0){
        num = num / 2;
    }

    while ((num % 3) == 0){
        num = num / 3;
    }

    while ((num % 5) == 0){
        num = num / 5;
    }

    return (num == 1);
}

public int addDigits(int num) {
    while (num > 9) {
        int sum = 0;
        while (num != 0) {
            int a = num % 10;
            sum += a;
            num = num / 10;
        }

        num = sum;
    }

    return num;
}
