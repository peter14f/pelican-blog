public int countNodes(TreeNode root) {
    if (root==null)
        return 0;

    int[] sum = {0};

    countNodes(sum, root);

    return sum[0];
}

private void countNodes(int[] sum, TreeNode node) {
    if (node == null)
        return;

    countNodes(sum, node.left);
    sum[0]++;
    countNodes(sum, node.right);
