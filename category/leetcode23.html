155-min_stack

Jan 04 2016 Tags Stacks

LeetCode Problem Link

We can use two stacks.

public class MinStack {
  Stack<Integer> stk;
  Stack<Integer> min;

  public MinStack() {
    stk = new Stack<Integer>();
    min = new Stack<Integer>();
  }

  public void push(int x) {
    stk.push(x);

    if (min.isEmpty() || x <= min.peek()) {
      min.push(x);
    }
  }

  public void pop() {
    int top = stk.peek();

    if (min.peek() == top)
      min.pop();

    stk.pop();
  }

  public int top() {
    return stk.peek();
  }

  public int getMin() {
    return min.peek();
  }
}

public int maxProduct(int[] nums) {

    int n = nums.length;
    int max = nums[0];

    for (int i=0; i<n; i++) {
        int product = nums[i];
        max = Math.max(max, product);

        for (int j=i+1; j<n; j++) {
            product = product * nums[j];
            max = Math.max(max, product);
        }
    }

    return max;
}
