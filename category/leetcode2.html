647-palindromic_substrings

Mar 19 2018 Tags Palindromes / Dyanamic Programming

LeetCode Problem Link

First thought is to go through all substrings starting and ending at each possible locations - this takes O(n^2) time. And then for each substring check if it's palindromic.

public int countSubstrings(String s) {

    char[] ch = s.toCharArray();
    int total = 0;

    // n^2 to go through all substrings
    for (int start=0; start<ch.length; start++) {
        for (int end=start; end<ch.length; end++) {
            if (isPalindrome(ch, start, end)) {
                total++;
            }
        }
    }

    return total;
}

private boolean isPalindrome(char[] ch, int start, int end) {
    while (start <= end) {
        if (ch[start] != ch[end]) {
            return false;
        }
        start++;
        end--;
    }
    return true;
}

public String tree2str(TreeNode t) {
    StringBuilder sb = new StringBuilder();

    if (t != null) {
        sb.append(t.val);
        if (t.left != null) {
            preOrderAppend(sb, t.left);
        } else if (t.right != null) {
            sb.append("()");
        }
        preOrderAppend(sb, t.right);
    }
    return sb.toString();
}

private void preOrderAppend(StringBuilder sb, TreeNode node) {
    if (node == null) {
        return;
    }

    sb.append('(');
    sb.append(node.val);

    if (node.left != null) {
        preOrderAppend(sb, node.left);
    } else if (node.right != null) {
        sb.append("()");
    }
    preOrderAppend(sb, node.right);

    sb.append(')');
}
