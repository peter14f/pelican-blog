460-lfu_cache

Mar 17 2018 Tags LFU / Doubly Linked Lists / Cache

LeetCode Problem Link

The Map<Integer, DDList> countMap tracks the list of nodes with the same frequency (usage count). It's good to clarify from the interviewer what the behavior should be when minCnt is bigger than 1. For LeetCode's solution, it does not feel truly LFU because we're asked to remove the last recently used item from the minCnt list and insert the element with a count of 1.

public class LFUCache {

static class Node {
    int key;
    int val;
    int cnt;
    Node prev, next;

    Node(int key, int val) {
        this.key = key;
        this.val = val;
        cnt = 1;
    }

    @Override
    public String toString() {
        return this.key + "," + this.val + ":" + this.cnt;
    }
}

static class DDList {
    Node head;
    Node tail;
    int size; // number of nodes in this list

    public DDList() {
        this.head = null;
        this.tail = null;
        this.size = 0;
    }

    public void insertAtHead(Node n) {
        n.next = this.head;
        if (this.head != null) {
            this.head.prev = n;
        } else {
            this.tail = n;
        }
        this.head = n;
        this.size++;
    }

    public Node removeFromTail() {
        if (this.tail == null) {
            return null;
        }

        Node oldTail = this.tail;
        if (this.tail.prev != null) {
            this.tail.prev.next = null;
        }

        this.tail = this.tail.prev;

        oldTail.prev = null;

        if (this.head == oldTail) {
            this.head = oldTail.next;
        }

        this.size--;
        return oldTail;
    }

    public Node removeNode(Node node) {
        if (node == null) {
            throw new IllegalArgumentException();
        }

        if (this.size == 1) {
            if (node != this.head || node != this.tail) {
                throw new IllegalArgumentException();
            }
            this.head = null;
            this.tail = null;
            this.size = 0;
            return node;
        } else {
            if (node.prev != null) {
                node.prev.next = node.next;
            }
            if (node.next != null) {
                node.next.prev = node.prev;
            }
            if (this.head == node) {
                this.head = node.next;
            }
            if (this.tail == node) {
                this.tail = node.prev;
            }
            this.size--;
            node.next = null;
            node.prev = null;
            return node;
        }
    }

    @Override
    public String toString() {
        Node cur = this.head;
        StringBuilder sb = new StringBuilder();
        while (cur != null) {
            sb.append(cur);
            sb.append(' ');
            cur = cur.next;
        }

        sb.append("|");

        cur = this.tail;
        while (cur != null) {
            sb.append(cur);
            sb.append(' ');
            cur = cur.prev;
        }
        return sb.toString();
    }
}

int capacity;
int size;
int minCnt; // current smallest frequency number
Map<Integer, Node> nodeMap;
Map<Integer, DDList> countMap; // head of a doubly-linked list

public LFUCache(int capacity) {
    this.capacity = capacity;
    this.nodeMap = new HashMap<>();
    this.countMap = new HashMap<>();
    this.minCnt = 1;
}

public int get(int key) {
    if (!this.nodeMap.containsKey(key)) {
        return -1;
    }
    Node node = this.nodeMap.get(key);
    int oldCnt = node.cnt;
    node.cnt++;

    this.countMap.get(oldCnt).removeNode(node);

    if (!this.countMap.containsKey(node.cnt)) {
        this.countMap.put(node.cnt, new DDList());
    }

    this.countMap.get(node.cnt).insertAtHead(node);

    if (this.minCnt == oldCnt && this.countMap.get(oldCnt).size == 0) {
        this.minCnt = node.cnt;
        this.countMap.remove(oldCnt);
    }

    return node.val;
}

public void put(int key, int value) {
    if (this.nodeMap.containsKey(key)) {
        Node node = this.nodeMap.get(key);
        node.val = value;
        int oldCnt = node.cnt;
        node.cnt++;
        this.countMap.get(oldCnt).removeNode(node);
        if (!this.countMap.containsKey(node.cnt)) {
            this.countMap.put(node.cnt, new DDList());
        }
        this.countMap.get(node.cnt).insertAtHead(node);
        if (this.minCnt == oldCnt && this.countMap.get(oldCnt).size == 0) {
            this.minCnt = node.cnt;
            this.countMap.remove(oldCnt);
        }
    } else {
        if (this.nodeMap.size() == this.capacity) {

            if (this.capacity == 0) {
                return;
            }

            Node removed = this.countMap.get(this.minCnt).removeFromTail();
            this.nodeMap.remove(removed.key);
        }
        Node newNode = new Node(key, value);
        this.nodeMap.put(key, newNode);
        if (!this.countMap.containsKey(newNode.cnt)) {
            this.countMap.put(newNode.cnt, new DDList());
        }
        this.countMap.get(newNode.cnt).insertAtHead(newNode);

        if (this.minCnt > 1) {
            this.minCnt = 1;
        }
    }
}

Remove(Node n) and removeFromTail() requires some care. Make sure all cases are handled correctly. For example, you could be removing from tail with a list of size 1. In that case, head needs to be updated as well.
