You should know the numbers for I, V, X. And L, C, D follow (LCD should be easy to remember). The last letter is M.

public int romanToInt(String s) {
    char[] s_arr = s.toCharArray();

    /* I -    1
     * V -    5
     * X -   10
     * L -   50
     * C -  100
     * D -  500
     * M - 1000
     *
     * 4 and 9
     */

    int sum = 0;

    for (int i=0; i<s_arr.length; i++) {

        int combo = nextCanBeSubtracted(s_arr, i);

        if (combo == 0) {
            sum += romanCharToInt(s_arr[i]);
        }
        else {
            sum += combo;
            i++;
        }
    }

    return sum;
}
