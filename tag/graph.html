public int longestIncreasingPath(int[][] matrix) {

    int[] ans = {0};

    int m = matrix.length;

    if (m == 0)
        return 0;

    int n = matrix[0].length;

    for (int row=0; row<m; row++) {
        for (int col=0; col<n; col++) {
            dfs(matrix, row, col, -1, -1, 1, ans);
        }
    }

    return ans[0];
}
