// 1  3  2  8  4  9

// 0  0  0  5  5  8

public int maxProfit(int[] prices, int fee) {
    int n = prices.length;

    int balanceWithStock = -prices[0];
    int balanceWithNoStock = 0;

    for (int i=1; i<n; i++) {
        balanceWithNoStock = Math.max(balanceWithNoStock, balanceWithStock + prices[i] - fee);
        balanceWithStock = Math.max(balanceWithStock, balanceWithNoStock - prices[i]);
    }

    return balanceWithNoStock;
}

The recursive solution is not hard to come up, but it takes exponential time.

public int findTargetSumWays(int[] nums, int S) {
    // 2 ^ n total possible choices
    return numWaysSumToTarget(nums, 0, 0, S);
}

private int numWaysSumToTarget(int[] nums, int i, int curSum, int target) {
    if (i >= nums.length) {
        if (curSum == target) {
            return 1;
        } else {
            return 0;
        }
    } else {
        int a = numWaysSumToTarget(nums, i+1, curSum+nums[i], target);
        int b = numWaysSumToTarget(nums, i+1, curSum-nums[i], target);
        return a + b;
    }
}
