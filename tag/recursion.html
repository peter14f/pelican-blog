public TreeNode invertTree(TreeNode root) {
    if (root == null)
        return null;

    invertTree(root.left);
    invertTree(root.right);

    TreeNode oldLeft = root.left;
    root.left = root.right;
    root.right = oldLeft;

    return root;
}
