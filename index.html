public int maxProfit(int[] prices, int fee) {
    int n = prices.length;

    int balanceWithStock = -prices[0];
    int balanceWithNoStock = 0;

    for (int i=1; i<n; i++) {
        balanceWithNoStock = Math.max(balanceWithNoStock, balanceWithStock + prices[i] - fee);
        balanceWithStock = Math.max(balanceWithStock, balanceWithNoStock - prices[i]);
    }

    return balanceWithNoStock;
}

public int hammingDistance(int x, int y) {
    int z = x ^ y;
    int hammingDistance = 0;
    for (int i=0; i<32; i++) {
        if ((z & 1) == 1) {
            hammingDistance++;
        }
        z = z >>> 1;
    }
    return hammingDistance;
}

477-total_hamming_distance

Jul 13 2018 Tags Couting Bits / Bit Manipulation

LeetCode Problem Link

Imagine each number is just zero or one, the total hamming distance is just (number of ones)*(number of zeros)

Do that for each bit and sum up the result at each bit.

public int totalHammingDistance(int[] nums) {
    int totalDistance = 0;

    for (int i=0; i<32; i++) {
        int numOnes = 0;
        int numZeros = 0;
        for (int j=0; j<nums.length; j++) {
            if ((nums[j] & 1) == 1) {
                numOnes++;
            } else {
                numZeros++;
            }
            nums[j] = nums[j] >>> 1;
        }
        totalDistance += (numOnes * numZeros);
    }

    return totalDistance;
}

LeetCode Problem Link

public int[][] transpose(int[][] A) {
    int m = A.length;

    if (m < 1) {
        return new int[0][0];
    }

    int n = A[0].length;

    int[][] B = new int[n][m];

    for (int i=0; i<m; i++) {
        for (int j=0; j<n; j++) {
            B[j][i] = A[i][j];
        }
    }

    return B;
}
