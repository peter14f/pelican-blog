public int countRangeSum(int[] nums, int lower, int upper) {
    int n = nums.length;

    int count = 0;

    for (int i=0; i<n; i++) {
        int curSum = nums[i];

        if (curSum >= lower && curSum <= upper)
            count++;

        for (int j=i+1; j<n; j++) {
            curSum += nums[j];

            if (curSum >= lower && curSum <= upper)
                count++;
        }
    }

    return count;
}
