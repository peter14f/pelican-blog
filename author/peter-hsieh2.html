public int countRangeSum(int[] nums, int lower, int upper) {
    int n = nums.length;

    int count = 0;

    for (int i=0; i<n; i++) {
        int curSum = nums[i];

        if (curSum >= lower && curSum <= upper)
            count++;

        for (int j=i+1; j<n; j++) {
            curSum += nums[j];

            if (curSum >= lower && curSum <= upper)
                count++;
        }
    }

    return count;
}

public int maxSubArrayLen(int[] nums, int k) {
    int n = nums.length;
    int maxSize = 0;

    for (int i=0; i<n; i++) {
        int sum = nums[i];

        if (sum == k && 1 > maxSize)
            maxSize = 1;

        for (int j=i+1; j<n; j++) {
            sum = sum + nums[j];

            if (sum == k && (j-i+1) > maxSize) {
                maxSize = j-i+1;
            }

        }
    }

    return maxSize;
