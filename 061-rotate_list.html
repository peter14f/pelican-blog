061-rotate_list

Oct 27 2015 Tags Linked Lists

LeetCode Problem Link

Calculate n in the first pass. Make tail's next point to the original head. And then determine the new head. And then set the new tail's next to null. Use modular arithmetic when k is bigger than n.

public ListNode rotateRight(ListNode head, int k) {

    if (head==null)
        return head;

    int n = 0;

    ListNode tail = head;

    while (tail != null) {
        n++;
        if (tail.next == null) {
            break;
        }
        tail = tail.next;
    }

    k = k - 1;
    if (k >= n) {
        k = k % n;
    }

    if (k==n-1)
        return head;

    tail.next = head;

    for (int i=0; i<n-k-1; i++) {
        head = head.next;
    }
    tail = head;
    for (int i=1; i<n; i++) {
        tail = tail.next;
    }
    tail.next = null;
    return head;
}
