The problem asks us to find the minimum number of commands. This should yell BFS. But this solution does not pass the OJ.

public int racecar(int target) {
    Queue<State> q = new LinkedList<>();
    Set<State> visited = new HashSet<>();

    State init = new State(1, 0);
    q.offer(init);
    visited.add(init);

    int numCommands = 0;
    while (!q.isEmpty()) {
        int size = q.size();

        for (int i=0; i<size; i++) {
            State cur = q.poll();
            //System.out.println(cur);

            // neighbor 1: A
            int newPosition = cur.position + cur.speed;
            int newSpeed = cur.speed * 2;
            State s1 = new State(newSpeed, newPosition);

            if (newPosition == target) {
                return numCommands + 1;
            } else if (!visited.contains(s1)) {
                q.offer(s1);
                visited.add(s1);
            }

            // neighbor 2: R
            newPosition = cur.position;
            newSpeed = (cur.speed > 0) ? -1 : 1;
            State s2 = new State(newSpeed, newPosition);

            if (newPosition == target) {
                return numCommands + 1;
            } else if (!visited.contains(s2)) {
                q.offer(s2);
                visited.add(s2);
            }
        }
        numCommands++;
    }

    return -1;
}

static class State {
    int speed = 0;
    int position = 0;

    public State(int speed, int position) {
        this.speed = speed;
        this.position = position;
    }

    @Override
    public int hashCode() {
        final int prime = 31;
        int result = 1;
        result = prime * result + position;
        result = prime * result + speed;
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj)
            return true;
        if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        State other = (State) obj;
        if (position != other.position)
            return false;
        if (speed != other.speed)
            return false;
        return true;
    }

    @Override
    public String toString() {
        return "speed: " + this.speed + " position: " + this.position;
    }
}
