dijkstra_algorithm

Mar 02 2016 Tags Dijkstra / Greedy / Graph / Weighted Graph / Undirected Graph / Directed Graph

In most psuedocode you will find for Dijkstra, they will say initialize the Q such that all nodes are put put in the Q.

The problem with this approach in Java is that when you update the distance to the node, PrioirtyQueue does not provide an API where you can update an item's priority. So you must remove that element and then re-insert it again.

class VertexComparator implements Comparator<Vertex> {

    @Override
    public int compare(Vertex o1, Vertex o2) {
        return o1.distance - o2.distance;
    }
}

class Vertex {
    int label;
    int distance;

    public Vertex (int label, int distance) {
        this.label = label;
        this.distance = distance;
    }
}

public int shortestDistance(int src, int dst, int n, int[][] edges) {

    List<Map<Integer, Integer>> graph = new ArrayList<Map<Integer, Integer>>();

    int srcNode = src - 'A';
    int dstNode = dst - 'A';

    Vertex[] distance = new Vertex[n];
    int k = edges.length;

    PriorityQueue<Vertex> pq = new PriorityQueue<Vertex>(new VertexComparator());
    Set<Integer> visited = new HashSet<Integer>();

    for (int i=0; i<n; i++) {
        // key is the neighbor, value is the weight of the edge
        graph.add(new HashMap<Integer, Integer>());

        Vertex v = null;
        if (i != srcNode)
            v = new Vertex(i, Integer.MAX_VALUE);
        else
            v = new Vertex(i, 0);

        distance[i] = v;

        pq.offer(v);
    }

    // build the graph
    for (int i=0; i<k; i++) {
        int a = edges[i][0] - 'A';
        int b = edges[i][1] - 'A';
        int weight = edges[i][2];

        graph.get(a).put(b, weight);
        graph.get(b).put(a, weight);
    }

    HashMap<Integer, Integer> prevNode = new HashMap<Integer, Integer>();

    while (!pq.isEmpty()) {
        Vertex v = pq.poll();
        int cur = v.label;
        System.out.println("cur=" + (char)(cur + 'A') + " " + distance[cur].distance);

        Map<Integer, Integer> neighbors = graph.get(cur);

        for (int neighbor: neighbors.keySet()) {
            System.out.println(" neighbor=" + (char)(neighbor + 'A'));
            int dist = neighbors.get(neighbor);

            if (dist + distance[cur].distance < distance[neighbor].distance) {
                System.out.println("  update " + (char)(neighbor + 'A'));
                prevNode.put(neighbor, cur);

                distance[neighbor].distance = dist + distance[cur].distance;
                pq.remove(distance[neighbor]);
                pq.add(distance[neighbor]);
            }

        }

        //System.out.println("adding " + (char)(cur + 'A'));
        visited.add(cur);

        if (cur == dstNode)
            break;
    }

    StringBuffer sb = new StringBuffer();
    int cur = dstNode;
    while (cur != srcNode) {
        sb.append((char)(cur+'A'));
        cur = prevNode.get(cur);
    }
    sb.append((char)(srcNode+'A'));

    System.out.println(sb.reverse().toString());

    return distance[dstNode].distance;
}

Or a better way to do start with only one node - the source node - in the PriorityQueue. You only put nodes into the priority queue when you have found a valid smaller distance.
