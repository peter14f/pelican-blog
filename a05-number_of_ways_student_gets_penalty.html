a05-number_of_ways_student_gets_penalty

Mar 19 2016 Tags counting problems

Attendance information has 3 possible states.

::
O: attendance L: late A: absence
If a student is "absent" for 2 days in total OR
"late" for 3 days in a row, he/she will get penalized.

There are 8 ways a student can get penalized in 3 days.

LLL AAO AAL OAA LAA AOA ALA AAA

N=3 -> ans=8 N=4 -> ans 38

P(consecutively late for 3...N days) + P(absent for a total of more than 2...N days)

public int numWays(int n) {
    int[] str = new int[n]; //0=A 1=O 2=L

    int cnt = 0;
    while (true) {
        if (isPenalizable(str)) {
            cnt++;
        }
        if (increment(str)) {
            break;
        }
    }

    return cnt;
}

private boolean isPenalizable(int[] str) {
    int numA = 0;
    int numConsecL = 0;
    for (int i=0; i<str.length; i++) {
        //0=A 1=O 2=L
        if (str[i] == 2) {
            numConsecL++;
            if (numConsecL == 3) {
                return true;
            }
        } else {
            numConsecL = 0;
            if (str[i] == 0) {
                numA++;
                if (numA == 2) {
                    return true;
                }
            }
        }
    }
    return false;
}

private boolean increment(int[] str) {
    str[0]++;

    boolean done = true;
    for (int i=0; i<str.length; i++) {
        if (str[i] <= 2) {
            done=false;
            break;
        } else {
            str[i] = 0;
            if (i+1 < str.length) {
                str[i+1]++;
            }
        }
    }
    return done;
}
